class ListingLoader {
    constructor() {
        this.sku = undefined;
        this.price = undefined;
        this.quantity = undefined;
        this.productId = undefined;
        this.productIdType = undefined;
        this.conditionType = undefined;
        this.conditionNote = undefined;
    }
    getFeedData() {
        if (!this.sku) throw new Error('No Sku');
        if (!this.price) throw new Error('No Price');
        if (!this.quantity && this.quantity !== 0) throw new Error('No Quantity');
        if (!this.productId) throw new Error('No Product Id');
        if (!this.productIdType) throw new Error('No Product Type');
        if (!this.conditionType) throw new Error('No Condition Type');
        if (!this.conditionNote && this.conditionNote !== '') throw new Error('No Condition Note');
        return `${this.sku}\t${this.price}\t${this.quantity}\t${this.productId}\t${this.productIdType}\t${this.conditionType}\t${this.conditionNote}																																															\n`
    }
}

async function main() {
    const x = new ListingLoader();
    x.sku = 'TestSKU';
    x.price = '14.99';
    x.quantity = 0;
    x.productId = 'B075ZFDXS4';
    x.productIdType = 'ASIN';
    x.conditionType = 'New';
    x.conditionNote = '';
    const feed = `${header}\n${x.getFeedData()}\n`;
    const fs = require('fs');
    fs.writeFileSync('test.txt', feed);
}

main();
